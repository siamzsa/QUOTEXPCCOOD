// Find any text node with "Demo Account" and change it to "Live Account" with styling document.querySelectorAll('*').forEach(function(node) { if (node.childNodes.length === 1 && node.childNodes[0].nodeType === 3) { if (node.textContent.trim() === 'Demo Account') { node.textContent = 'Live Account'; // Apply style: green color, bold, 14px font size node.style.color = "#0faf59"; node.style.fontWeight = "bold"; node.style.fontSize = "10px"; } } });
